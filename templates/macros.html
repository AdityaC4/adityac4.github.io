{# Reusable template macros #} {% macro post_link(path, text, attrs="") %} {# Resolve the page and render a safe link; attrs is an optional string of HTML attributes #} {% set p = get_page(path=path) %} {% if p %} {% if attrs %} {{ text }} {% else %} {{ text }} {% endif %} {% else %} {# Print text and include a small HTML comment for debugging when missing #} {{ text }} {% endif %} {% endmacro %}
